in order to match with the new forum and to allow acl editing, I had to 
change a few messages to forumadmin :

# forum admin
btn.create = Create
btn.remove = Remove
btn.accessControl = Access Control
btn.close = Close
msg.newForum = New Forum

err.list = Unable to fetch the forum list
err.create = Unable to create the forum
err.delete = Unable to delete the forum

# access control
msg.accessControl = Access Control :
btn.addGroup = Add Group
btn.addUser = Add User
lbl.allow = Allow
lbl.deny = Deny

msg.allowGroup = Allow group :
msg.allowUser = Allow user :
msg.denyGroup = Deny group :
msg.denyUser = Deny user :
msg.selectGroup = Select a group :
msg.selectUser = Select a user :

err.unableToReadAccessControl = Unable to read access control list
err.unableToListGroups = Unable to read the list of groups :
err.unableToListUsers = Unable to read the list of users :
err.unableToAddGroup = Unable to add this group :
err.unableToAddUser = Unable to add this user :
err.unableToRemoveAccessRule = Unable to remove this access rule :
